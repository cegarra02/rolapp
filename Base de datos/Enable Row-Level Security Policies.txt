-- =============================================
-- CHARACTERS_LIBRARY
-- =============================================

-- Cualquiera puede ver personajes aprobados (sin login)
create policy "cl_select_approved"
on public.characters_library for select
using (status = 'approved');

-- Admin ve todos los personajes (para moderar)
create policy "cl_admin_select_all"
on public.characters_library for select
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);

-- Usuarios autenticados pueden insertar personajes aprobados
create policy "cl_insert_own"
on public.characters_library for insert
to authenticated
with check (true);

-- Admin puede actualizar cualquier personaje (aprobar, cambiar status, etc.)
create policy "cl_admin_update"
on public.characters_library for update
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);

-- =============================================
-- SUBMISSIONS
-- =============================================

-- Cada usuario ve sus propias submissions
create policy "sub_select_own"
on public.submissions for select
to authenticated
using (author_id = auth.uid());

-- Admin ve todas las submissions
create policy "sub_admin_select_all"
on public.submissions for select
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);

-- Usuarios autenticados pueden enviar submissions
create policy "sub_insert_own"
on public.submissions for insert
to authenticated
with check (author_id = auth.uid());

-- Admin puede cambiar el estado (aprobar/rechazar)
create policy "sub_admin_update"
on public.submissions for update
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);

-- =============================================
-- USERS
-- =============================================

-- Cada usuario ve su propia fila
create policy "users_select_own"
on public.users for select
to authenticated
using (id = auth.uid());

-- Admin ve cualquier fila (necesario para leer gemas del autor al aprobar submissions)
create policy "users_admin_select_all"
on public.users for select
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);

-- Cada usuario crea su propia fila al registrarse
create policy "users_insert_own"
on public.users for insert
to authenticated
with check (id = auth.uid());

-- Solo el admin puede modificar gemas (nadie puede inflarse gemas desde el cliente)
create policy "users_admin_update"
on public.users for update
to authenticated
using (
  (select email from auth.users where id = auth.uid()) = 'alex1234567890ct@gmail.com'
);
